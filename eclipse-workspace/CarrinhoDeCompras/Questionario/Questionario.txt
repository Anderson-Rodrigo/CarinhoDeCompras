public interface Loja{
	String Compra();
}

public class Usuario
	implements Loja{

	private String nome;
	
	public Nome(String nome){
		this.nome = nome;
	}

	public String getNome() {
		return nome;
	}
}

public class Compras
	implements Loja{

	private String produto;
	
	public produt(String produt){
		this.produto = produto;
	}

	public String getproduto() {
		return produt;
	}
}
